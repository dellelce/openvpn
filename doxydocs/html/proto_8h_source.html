176 /*
177  * The following macro is used to update an
178  * internet checksum. "acc" is a 32-bit
179  * accumulation of all the changes to the
180  * checksum (adding in old 16-bit words and
181  * subtracting out new words), and "cksum"
182  * is the checksum value to be updated.
183  */
184 #define ADJUST_CHECKSUM(acc, cksum) { \
185  int _acc = acc; \
186  _acc += (cksum); \
187  if (_acc < 0) { \
188  _acc = -_acc; \
189  _acc = (_acc >> 16) + (_acc & 0xffff); \
190  _acc += _acc >> 16; \
191  (cksum) = (uint16_t) ~_acc; \
192  } else { \
193  _acc = (_acc >> 16) + (_acc & 0xffff); \
194  _acc += _acc >> 16; \
195  (cksum) = (uint16_t) _acc; \
196  } \
197 }
198 
199 #define ADD_CHECKSUM_32(acc, u32) { \
200  acc += (u32) & 0xffff; \
201  acc += (u32) >> 16; \
202 }
203 
204 #define SUB_CHECKSUM_32(acc, u32) { \
205  acc -= (u32) & 0xffff; \
206  acc -= (u32) >> 16; \

220 /*
221  * This returns an ip protocol version of packet inside tun
222  * and offset of IP header (via parameter).
223  */
224 inline static int
225 get_tun_ip_ver(int tunnel_type, struct buffer *buf, int *ip_hdr_offset)
226 {
227  int ip_ver = -1;
228 
229  /* for tun get ip version from ip header */
230  if (tunnel_type == DEV_TYPE_TUN)
231  {
232  *ip_hdr_offset = 0;
233  if (likely(BLEN(buf) >= (int) sizeof(struct openvpn_iphdr)))
234  {
235  ip_ver = OPENVPN_IPH_GET_VER(*BPTR(buf));
236  }
237  }
238  else if (tunnel_type == DEV_TYPE_TAP)
239  {
240  *ip_hdr_offset = (int)(sizeof(struct openvpn_ethhdr));
241  /* for tap get ip version from eth header */
242  if (likely(BLEN(buf) >= *ip_hdr_offset))
243  {
244  const struct openvpn_ethhdr *eh = (const struct openvpn_ethhdr *) BPTR(buf);
245  uint16_t proto = ntohs(eh->proto);
246  if (proto == OPENVPN_ETH_P_IPV6)
247  {
248  ip_ver = 6;
249  }
250  else if (proto == OPENVPN_ETH_P_IPV4)
251  {
252  ip_ver = 4;
253  }
254  }
255  }
256 
257  return ip_ver;
258 }
