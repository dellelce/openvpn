368 static inline in_addr_t
369 netbits_to_netmask(const int netbits)
370 {
371  const int addrlen = sizeof(in_addr_t) * 8;
372  in_addr_t mask = 0;
373  if (netbits > 0 && netbits <= addrlen)
374  {
375  mask = IPV4_NETMASK_HOST << (addrlen-netbits);
376  }
377  return mask;
378 }
