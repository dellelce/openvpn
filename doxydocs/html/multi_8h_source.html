562 /*
563  * Compute earliest timeout expiry from the set of
564  * all instances. Output:
565  *
566  * m->earliest_wakeup : instance needing the earliest service.
567  * dest : earliest timeout as a delta in relation
568  * to current time.
569  */
570 static inline void
571 multi_get_timeout(struct multi_context *m, struct timeval *dest)
572 {
573  struct timeval tv, current;
574 
575  CLEAR(tv);
576  m->earliest_wakeup = (struct multi_instance *) schedule_get_earliest_wakeup(m->schedule, &tv);
577  if (m->earliest_wakeup)
578  {
579  ASSERT(!openvpn_gettimeofday(&current, NULL));
580  tv_delta(dest, &current, &tv);
581  if (dest->tv_sec >= REAP_MAX_WAKEUP)
582  {
583  m->earliest_wakeup = NULL;
584  dest->tv_sec = REAP_MAX_WAKEUP;
585  dest->tv_usec = 0;
586  }
587  }
588  else
589  {
590  dest->tv_sec = REAP_MAX_WAKEUP;
591  dest->tv_usec = 0;
592  }
593 }
